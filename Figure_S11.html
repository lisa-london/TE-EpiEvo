ana3_melted2 <-read.csv("GitHubData/S10_singleton_activity.csv")

ana3_melted2$DNMT <- as.factor(ana3_melted2$DNMT)
n_fun <- function(x){
  return(data.frame(y = 0.95*800,
                    label = length(x)))}
or_dnmts2<-ggplot(ana3_melted2,aes(x=variable,y=value,fill=DNMT))+
    stat_boxplot(geom ='errorbar', width = 0.6) +geom_boxplot(width = 0.6)+
    theme_bw()+
    theme(text=element_text(size=16,family="Times New Roman"),legend.position='bottom')+
    scale_fill_manual(values=c('#beaed4','#7fc97f'),name='',labels=c('DNMTs absent','DMMTs present'))+
    stat_compare_means(aes(label = paste0("p = ", ..p.format..)),method='wilcox.test',
                       label.x.npc='left', label.y.npc=0.9,size=5,family="Times New Roman")+
    expand_limits(y = 0)+
    ylab("Number of Singletons")+xlab("Singleton TE Activity")+
    stat_summary(fun.data = n_fun, geom = "text", aes(group=DNMT),family="Times New Roman",
                 hjust = 0.5, position = position_dodge(0.6))
